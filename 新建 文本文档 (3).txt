#include <wiringPi.h>
#include <iostream>

#define LED_PIN 18 //设置GPIO18为LED控制引脚

int main()
{
    if(wiringPiSetupGpio() == -1){ //初始化wiringPi库
        std::cerr << "Failed to initialize wiringPi" << std::endl;
        return -1;
    }

    pinMode(LED_PIN, OUTPUT); //将LED控制引脚设置为输出模式

    while(1){
        digitalWrite(LED_PIN, HIGH); //设置LED控制引脚输出高电平
        delay(1000); //延迟1秒

        digitalWrite(LED_PIN, LOW); //设置LED控制引脚输出低电平
        delay(1000); //延迟1秒
    }

    return 0;
}
